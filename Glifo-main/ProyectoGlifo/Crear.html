3 Configuración del juego
Se definen las constantes del juego: tamaño del tablero (9x9), número de minas (10), y las variables globales que controlan el estado del juego. 
const width = 9;      // 9 columnas
const height = 9;     // 9 filas
const totalMines = 10; // 10 minas en total

let board = [];       // Tablero lógico (valores)
let cells = [];       // Referencias a las celdas HTML
let gameActive = true;
let firstClick = true;
let revealedCount = 0;
let minesMarked = 0;
let timer = 0;
let timerInterval = null;
4 Crear el tablero HTML
La función createBoardHTML() genera dinámicamente las 81 celdas del tablero. A cada celda se le asigna un evento de clic izquierdo (revelar) y clic derecho (marcar bandera). 
function createBoardHTML() {
    const boardElement = document.getElementById('board');
    boardElement.innerHTML = '';
    cells = [];
    
    for (let row = 0; row < height; row++) {
        for (let col = 0; col < width; col++) {
            const cell = document.createElement('div');
            cell.className = 'cell';
            cell.addEventListener('click', () => handleClick(row, col));
            cell.addEventListener('contextmenu', (e) => {
                e.preventDefault();
                handleRightClick(row, col);
            });
            boardElement.appendChild(cell);
            cells.push(cell);
        }
    }
}
5 Colocar minas (seguridad del primer clic)
Las minas se colocan DESPUÉS del primer clic del jugador. Esto asegura que el jugador nunca pierda en el primer movimiento, mejorando la experiencia de juego. 
function placeMines(firstRow, firstCol) {
    let minesPlaced = 0;
    while (minesPlaced < totalMines) {
        const row = Math.floor(Math.random() * height);
        const col = Math.floor(Math.random() * width);
        if ((row === firstRow && col === firstCol) || board[row][col] === -1) continue;
        board[row][col] = -1;
        minesPlaced++;
    }
    calculateNumbers();
}
6 Calcular números de minas adyacentes
Después de colocar las minas, se recorre cada celda y se cuenta cuántas minas hay en las 8 posiciones alrededor. Este número se almacena en el tablero lógico. 
function calculateNumbers() {
    for (let row = 0; row < height; row++) {
        for (let col = 0; col < width; col++) {
            if (board[row][col] === -1) continue;
            let mineCount = 0;
            for (let dr = -1; dr <= 1; dr++) {
                for (let dc = -1; dc <= 1; dc++) {
                    const newRow = row + dr;
                    const newCol = col + dc;
                    if (newRow >= 0 && newRow < height && newCol >= 0 && newCol < width) {
                        if (board[newRow][newCol] === -1) mineCount++;
                    }
                }
            }
            board[row][col] = mineCount;
        }
    }
}
7 Revelar celdas (efecto cascada)
Cuando se revela una celda con valor 0 (sin minas alrededor), automáticamente se revelan todas las celdas adyacentes recursivamente. Esto crea el efecto "cascada" característico del Buscaminas. 
function revealCell(row, col, isAutoReveal = false) {
    const cell = cells[row * width + col];
    const value = board[row][col];
    
    if (value === -1) {
        cell.classList.add('revealed', 'mine');
        gameActive = false;
        showMessage('💥 ¡HAS PERDIDO! 💥', 'lose');
        revealAllMines();
        return;
    }
    
    cell.classList.add('revealed');
    revealedCount++;
    
    if (value > 0) {
        cell.textContent = value;
        cell.setAttribute('data-value', value);
    } else if (value === 0) {
        for (let dr = -1; dr <= 1; dr++) {
            for (let dc = -1; dc <= 1; dc++) {
                if (dr === 0 && dc === 0) continue;
                const newRow = row + dr;
                const newCol = col + dc;
                const neighborCell = cells[newRow * width + newCol];
                if (neighborCell && !neighborCell.classList.contains('revealed')) {
                    revealCell(newRow, newCol, true);
                }
            }
        }
    }
}
8 Manejo de clics (izquierdo y derecho)
El clic izquierdo revela la celda. El clic derecho marca o desmarca una bandera (🚩) para señalar dónde el jugador cree que hay una mina. 
function handleClick(row, col) {
    if (!gameActive) return;
    if (firstClick) {
        firstClick = false;
        placeMines(row, col);
        startTimer();
    }
    revealCell(row, col);
}

function handleRightClick(row, col) {
    if (!gameActive || firstClick) return;
    const cell = cells[row * width + col];
    if (cell.classList.contains('revealed')) return;
    
    if (cell.classList.contains('flagged')) {
        cell.classList.remove('flagged');
        cell.textContent = '';
        minesMarked--;
    } else {
        cell.classList.add('flagged');
        cell.textContent = '🚩';
        minesMarked++;
    }
    updateMineCounter();
}
9 Verificar victoria y reinicio
El jugador gana cuando ha revelado todas las celdas que NO contienen minas. El botón de reinicio permite comenzar una nueva partida. 
function checkWin() {
    const totalSafeCells = (width * height) - totalMines;
    if (revealedCount === totalSafeCells) {
        gameActive = false;
        showMessage('🎉 ¡FELICIDADES! HAS GANADO 🎉', 'win');
        stopTimer();
    }
}

function resetGame() {
    initGame();
}

document.getElementById('resetButton').addEventListener('click', () => resetGame());
📝 Resumen del funcionamiento
1. El tablero es una matriz de 9x9 (81 celdas).
2. Las minas se colocan aleatoriamente después del primer clic.
3. Cada celda almacena un número (0-8) que indica cuántas minas hay alrededor.
4. Al hacer clic en una celda con 0, se revelan automáticamente las adyacentes.
5. Clic derecho marca banderas para señalar posibles minas.
6. El juego termina cuando se pisa una mina (pierdes) o se revelan todas las celdas seguras (ganas).

<body>
    <div class="container">
        <h1>💣 BUSCAMINAS</h1>
        
        <div class="game-controls">
            <div class="mine-counter">💣 <span id="mineCounter">10</span></div>
            <button id="resetButton">🔄 Nuevo juego</button>
            <div class="timer">⏱️ <span id="timer">000</span></div>

    <script>
        const width = 9;
        const height = 9;
        const totalMines = 10;

        let board = [];
        let cells = [];
        let gameActive = true;
        let firstClick = true;
        let revealedCount = 0;
        let minesMarked = 0;
        let timer = 0;
        let timerInterval = null;

        function initGame() {
            board = Array(height).fill().map(() => Array(width).fill(0));
            firstClick = true;
            gameActive = true;
            revealedCount = 0;
            minesMarked = 0;
            
            if (timerInterval) {
                clearInterval(timerInterval);
                timerInterval = null;
            }
            timer = 0;
            updateTimerDisplay();
            updateMineCounter();
            hideMessage();
            createBoardHTML();
        }

        function createBoardHTML() {
            const boardElement = document.getElementById('board');
            boardElement.innerHTML = '';
            cells = [];
            
            for (let row = 0; row < height; row++) {
                for (let col = 0; col < width; col++) {
                    const cell = document.createElement('div');
                    cell.className = 'cell';
                    cell.dataset.row = row;
                    cell.dataset.col = col;
                    cell.addEventListener('click', () => handleClick(row, col));
                    cell.addEventListener('contextmenu', (e) => {
                        e.preventDefault();
                        handleRightClick(row, col);
                    });
                    boardElement.appendChild(cell);
                    cells.push(cell);
                }
            }
        }

        function placeMines(firstRow, firstCol) {
            let minesPlaced = 0;
            while (minesPlaced < totalMines) {
                const row = Math.floor(Math.random() * height);
                const col = Math.floor(Math.random() * width);
                if ((row === firstRow && col === firstCol) || board[row][col] === -1) continue;
                board[row][col] = -1;
                minesPlaced++;
            }
            calculateNumbers();
        }

        function calculateNumbers() {
            for (let row = 0; row < height; row++) {
                for (let col = 0; col < width; col++) {
                    if (board[row][col] === -1) continue;
                    let mineCount = 0;
                    for (let dr = -1; dr <= 1; dr++) {
                        for (let dc = -1; dc <= 1; dc++) {
                            const newRow = row + dr;
                            const newCol = col + dc;
                            if (newRow >= 0 && newRow < height && newCol >= 0 && newCol < width) {
                                if (board[newRow][newCol] === -1) mineCount++;
                            }
                        }
                    }
                    board[row][col] = mineCount;
                }
            }
        }

        function revealCell(row, col, isAutoReveal = false) {
            if (row < 0 || row >= height || col < 0 || col >= width) return;
            const cell = cells[row * width + col];
            const value = board[row][col];
            
            if (!isAutoReveal && (cell.classList.contains('revealed') || cell.classList.contains('flagged'))) return;
            
            if (value === -1) {
                cell.classList.add('revealed', 'mine');
                gameActive = false;
                showMessage('💥 ¡HAS PERDIDO! 💥', 'lose');
                revealAllMines();
                stopTimer();
                return;
            }
            
            cell.classList.add('revealed');
            revealedCount++;
            
            if (value > 0) {
                cell.textContent = value;
                cell.setAttribute('data-value', value);
            } else if (value === 0) {
                cell.textContent = '';
                for (let dr = -1; dr <= 1; dr++) {
                    for (let dc = -1; dc <= 1; dc++) {
                        if (dr === 0 && dc === 0) continue;
                        const newRow = row + dr;
                        const newCol = col + dc;
                        const neighborCell = cells[newRow * width + newCol];
                        if (neighborCell && !neighborCell.classList.contains('revealed')) {
                            revealCell(newRow, newCol, true);
                        }
                    }
                }
            }
            checkWin();
        }

        function handleClick(row, col) {
            if (!gameActive) return;
            if (firstClick) {
                firstClick = false;
                placeMines(row, col);
                startTimer();
            }
            const cell = cells[row * width + col];
            if (cell.classList.contains('flagged')) return;
            revealCell(row, col);
        }

        function handleRightClick(row, col) {
            if (!gameActive || firstClick) return;
            const cell = cells[row * width + col];
            if (cell.classList.contains('revealed')) return;
            
            if (cell.classList.contains('flagged')) {
                cell.classList.remove('flagged');
                cell.textContent = '';
                minesMarked--;
            } else {
                cell.classList.add('flagged');
                cell.textContent = '🚩';
                minesMarked++;
            }
            updateMineCounter();
        }

        function revealAllMines() {
            for (let row = 0; row < height; row++) {
                for (let col = 0; col < width; col++) {
                    if (board[row][col] === -1) {
                        const cell = cells[row * width + col];
                        if (!cell.classList.contains('flagged')) {
                            cell.classList.add('revealed', 'mine');
                        }
                    }
                }
            }
        }

        function checkWin() {
            const totalSafeCells = (width * height) - totalMines;
            if (revealedCount === totalSafeCells) {
                gameActive = false;
                showMessage('🎉 ¡FELICIDADES! HAS GANADO 🎉', 'win');
                stopTimer();
            }
        }

        function startTimer() {
            timerInterval = setInterval(() => {
                timer++;
                updateTimerDisplay();
            }, 1000);
        }

        function stopTimer() {
            if (timerInterval) {
                clearInterval(timerInterval);
                timerInterval = null;
            }
        }

        function updateTimerDisplay() {
            const timerElement = document.getElementById('timer');
            if (timerElement) timerElement.textContent = timer.toString().padStart(3, '0');
        }

        function updateMineCounter() {
            const remainingMines = totalMines - minesMarked;
            const counterElement = document.getElementById('mineCounter');
            if (counterElement) counterElement.textContent = remainingMines.toString().padStart(2, '0');
        }

        function showMessage(msg, type) {
            const messageElement = document.getElementById('message');
            if (messageElement) {
                messageElement.textContent = msg;
                messageElement.className = `message ${type}`;
            }
        }

        function hideMessage() {
            const messageElement = document.getElementById('message');
            if (messageElement) {
                messageElement.textContent = '';
                messageElement.className = 'message';
            }
        }

        function resetGame() {
            initGame();
        }

        document.getElementById('resetButton').addEventListener('click', () => resetGame());
        initGame();
    </script>
